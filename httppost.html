                a.改头文件
                $http({
                    method: 'POST',
                    url: './services/post.php',
                    data: {
                        a: $scope.a,
                        b: $scope.b
                    },
                    headers: {
                        'content-type': 'application/x-www-form-urlencoded'
                    },
                    transformRequest: function (obj) {
                        let arr=[];
                        for(let name in obj) {
                            arr.push(`${name}=${obj[name]}`);
                        }
                        return arr.join("&");
                    }
                })
                .then(res => {
                    alert(res.data);
                }, () => {
                    alert("Error");
                });
